const express = require('express'); const session = require('express-session'); const multer = require('multer'); const bcrypt = require('bcryptjs'); const path = require('path'); const fs = require('fs'); const app = express(); const PORT = process.env.PORT || 3000; // Ensure uploads dir exists const uploadsDir = path.join(__dirname, 'uploads'); if (!fs.existsSync(uploadsDir)) fs.mkdirSync(uploadsDir); // Config file for storing ticket password & PDF info const configPath = path.join(__dirname, 'config.json'); function getConfig() { if (!fs.existsSync(configPath)) { return { ticketPassword: null, ticketFile: null, ticketName: null, welcomeMessage: 'Κάτι σε περιμένει...' }; } return JSON.parse(fs.readFileSync(configPath, 'utf8')); } function saveConfig(data) { fs.writeFileSync(configPath, JSON.stringify(data, null, 2)); } // Multer storage const storage = multer.diskStorage({ destination: (req, file, cb) => cb(null, uploadsDir), filename: (req, file, cb) => cb(null, 'ticket.pdf') }); const upload = multer({ storage, fileFilter: (req, file, cb) => { if (file.mimetype === 'application/pdf') cb(null, true); else cb(new Error('Only PDFs allowed')); }}); // Middleware app.use(express.json()); app.use(express.urlencoded({ extended: true })); app.use(express.static(path.join(__dirname, 'public'))); app.use(session({ secret: process.env.SESSION_SECRET || 'supersecretkey_changeme', resave: false, saveUninitialized: false, cookie: { maxAge: 24 * 60 * 60 * 1000 } })); // ─── GUEST ROUTES ─────────────────────────────────────────── // Main guest page app.get('/', (req, res) => { res.sendFile(path.join(__dirname, 'public', 'index.html')); }); // Verify anniversary date app.post('/api/verify', (req, res) => { const { code } = req.body; const config = getConfig(); if (!config.ticketPassword) { return res.json({ success: false, message: 'Δεν έχει οριστεί κωδικός ακόμα.' }); } // Compare (case-insensitive, strip spaces/dashes) const normalize = s => s.replace(/[\s\-\/\.]/g, '').toLowerCase(); if (normalize(code) === normalize(config.ticketPassword)) { req.session.unlocked = true; return res.json({ success: true }); } res.json({ success: false, message: 'Λάθος ημερομηνία. Σκέψου καλύτερα... 💭' }); }); // Serve the PDF (only if unlocked) app.get('/api/ticket', (req, res) => { if (!req.session.unlocked) { return res.status(403).json({ error: 'Unauthorized' }); } const config = getConfig(); if (!config.ticketFile) { return res.status(404).json({ error: 'No ticket uploaded yet' }); } const filePath = path.join(uploadsDir, 'ticket.pdf'); if (!fs.existsSync(filePath)) { return res.status(404).json({ error: 'File not found' }); } res.setHeader('Content-Type', 'application/pdf'); res.setHeader('Content-Disposition', 'inline; filename="ticket.pdf"'); res.sendFile(filePath); }); // Get welcome message & whether ticket is set app.get('/api/info', (req, res) => { const config = getConfig(); res.json({ welcomeMessage: config.welcomeMessage || 'Κάτι σε περιμένει...', hasTicket: !!config.ticketFile, unlocked: !!req.session.unlocked }); }); // ─── ADMIN ROUTES ──────────────────────────────────────────── // Admin password check middleware const ADMIN_PASSWORD = process.env.ADMIN_PASSWORD || 'admin123'; function requireAdmin(req, res, next) { if (req.session.isAdmin) return next(); res.status(401).json({ error: 'Not authenticated' }); } app.get('/admin', (req, res) => { res.sendFile(path.join(__dirname, 'public', 'admin.html')); }); app.post('/api/admin/login', (req, res) => { const { password } = req.body; if (password === ADMIN_PASSWORD) { req.session.isAdmin = true; return res.json({ success: true }); } res.json({ success: false, message: 'Λάθος κωδικός.' }); }); app.post('/api/admin/logout', (req, res) => { req.session.isAdmin = false; res.json({ success: true }); }); app.get('/api/admin/status', requireAdmin, (req, res) => { const config = getConfig(); res.json({ config }); }); app.post('/api/admin/upload', requireAdmin, upload.single('ticket'), (req, res) => { const { ticketPassword, welcomeMessage } = req.body; if (!req.file) return res.json({ success: false, message: 'Δεν ανέβηκε αρχείο.' }); if (!ticketPassword) return res.json({ success: false, message: 'Βάλε κωδικό ξεκλειδώματος.' }); const config = getConfig(); config.ticketFile = 'ticket.pdf'; config.ticketName = req.file.originalname; config.ticketPassword = ticketPassword; config.welcomeMessage = welcomeMessage || config.welcomeMessage; saveConfig(config); res.json({ success: true, message: 'Το εισητήριο ανέβηκε! 🎫' }); }); app.post('/api/admin/update-settings', requireAdmin, (req, res) => { const { ticketPassword, welcomeMessage } = req.body; const config = getConfig(); if (ticketPassword) config.ticketPassword = ticketPassword; if (welcomeMessage) config.welcomeMessage = welcomeMessage; saveConfig(config); res.json({ success: true, message: 'Οι ρυθμίσεις αποθηκεύτηκαν.' }); }); app.delete('/api/admin/ticket', requireAdmin, (req, res) => { const config = getConfig(); const filePath = path.join(uploadsDir, 'ticket.pdf'); if (fs.existsSync(filePath)) fs.unlinkSync(filePath); config.ticketFile = null; config.ticketName = null; saveConfig(config); res.json({ success: true }); }); app.listen(PORT, () => { console.log(`✅ Server running on http://localhost:${PORT}`); console.log(`🔑 Admin password: ${ADMIN_PASSWORD}`); });
